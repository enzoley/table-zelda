MPD

CREATE TABLE GENRE(
ID INT PRIMARY KEY,
NOM VARCHAR(40) NOT NULL,
DESCRIPTION VARCHAR(40)
);

CREATE TABLE THEME(
ID INT PRIMARY KEY,
NOM VARCHAR(40) NOT NULL,
description VARCHAR(40)
);

CREATE TABLE CONSOLE(
ID INT PRIMARY KEY,
NOM VARCHAR(40) NOT NULL,
DATE_DE_SORTIE DATE,
IMPACT_ENVIRONEMENTAL VARCHAR(40),
TYPE VARCHAR(40) NOT NULL
);

CREATE TABLE JOUEUR(
ID INT PRIMARY KEY,
PSEUDO VARCHAR(40) NOT NULL,
DATE_DE_NAISSANCE DATE NOT NULL
);

CREATE TABLE EDITEUR(
ID INT PRIMARY KEY,
NOM VARCHAR(40) NOT NULL,
LOCALISATION VARCHAR(40) NOT NULL,
DATE_DE_CREATION DATE NOT NULL,
FONDATEUR VARCHAR(40) NOT NULL
);

CREATE TABLE CLASSIFICATION(
ID INT PRIMARY KEY,
AGE INT NOT NULL CHECK
(AGE >= 0 AND AGE <= 18)
);

CREATE TABLE JEU(
ID INT PRIMARY KEY,
TITRE VARCHAR(40) NOT NULL,
DATE_DE_SORTIE DATE,
NOTE_JOURNALISTIQUE INT,
MULTIJOUEUR BOOLEAN,
THEME INT,
CLASSIFICATION INT,
FOREIGN KEY(THEME) REFERENCES THEME(ID),
FOREIGN KEY(CLASSIFICATION) REFERENCES CLASSIFICATION(ID)
);

CREATE TABLE NOTER(
IDJOUEUR INT,
IDJEU INT,
NOTE INT,
PRIMARY KEY(IDJOUEUR, IDJEU)
FOREIGN KEY(IDJOUEUR) REFERENCES JOUEUR(ID),
FOREIGN KEY(IDJEU) REFERENCES JEU(ID)
);

CREATE TABLE EST_GENRE(
IDGENRE INT,
IDJEU INT,
PRIMARY KEY (IDGENRE,IDJEU),
FOREIGN KEY(IDGENRE) REFERENCES GENRE(ID),
FOREIGN KEY(IDJEU) REFERENCES JEU(ID)
);

CREATE TABLE JOUE_SUR(
IDJEU INT,
IDCONSOLE INT,
PRIMARY KEY (IDJEU,IDCONSOLE),
FOREIGN KEY(IDJEU) REFERENCES JEU(ID),
FOREIGN KEY(IDCONSOLE) REFERENCES CONSOLE(ID)
);

CREATE TABLE CREER_PAR(
IDJEU INT,
IDEDITEUR INT,
PRIMARY KEY(IDJEU,IDEDITEUR),
FOREIGN KEY(IDJEU) REFERENCES JEU(ID),
FOREIGN KEY(IDEDITEUR) REFERENCES EDITEUR(ID)
)
